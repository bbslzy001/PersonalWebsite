DML常用命令

  • 插入数据
    • 插入单个元组
      mysql> insert into s values('95020','陈东','男',18,'IS',null);
    • 插入多个元组
      mysql> insert into s(sno,sname) values          
    -> ('1','li'),
    -> ('2','zhang'),
    -> ('3','wang');
    • 插入子查询结果
      • 将子查询结果插入指定表中
      #建表#
mysql> create table dd(sdept char(15), average int);
#插入数据#
mysql> insert into dd(sdept,average)
    -> select sdept,avg(sage) from s group by sdept;
  • 修改数据
    • 修改一个元组的值
      mysql> update s set sage=22 where sno='1';
    • 修改多个元组的值
      mysql> update s set sage =sage+1;
    • 带子查询的修改
  • 删除数据
    • 删除一个元组的值
      mysql> delete from s where sno='1';
    • 删除多个元组的值
      mysql> delete from sc;

#如果要删除null所在列，不能使用=，使用is null或者not is null#
mysql> delete from s where sage is null;
    • 带子查询的删除
