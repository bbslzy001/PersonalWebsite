差分

一维差分

给区间[l, r]中的每个数加上c：B[l] += c, B[r + 1] -= c

二维差分

给以(x1, y1)为左上角，(x2, y2)为右下角的子矩阵中的所有元素加上c： S[x1, y1] += c, S[x2 + 1, y1] -= c, S[x1, y2 + 1] -= c, S[x2 + 1, y2 + 1] += c
#include <cstdio>

int main(void)
{
	int n, m;
	scanf("%d%d", &n, &m);
	int* a = new int[n+1]();
	int temp;
	for (int i = 1; i <= n; ++i)
	{
		scanf("%d", &temp);
		a[i] += temp;
		a[i + 1] -= temp;
	}
	while (m--)
	{
		int l, r, c;
		scanf("%d%d%d", &l, &r, &c);
		a[l] += c;
		a[r + 1] -= c;
	}
	for (int i = 1; i <= n; ++i)
	{
		a[i] += a[i - 1];
		printf("%d ", a[i]);
	}

	return 0;
}

二位差分数组：

#include <cstdio>

const int N = 1010;
int ar[N][N] = {};
int origin[N][N] = {};

int main(void)
{
	int n, m, q;
	scanf("%d%d%d", &n, &m, &q);
	for (int i = 1; i <= n; ++i)
	{
		for (int j = 1; j <= m; ++j)
		{
			scanf("%d", &origin[i][j]);
		}
	}
	while (q--)
	{
		int x1, y1, x2, y2, c;
		scanf("%d%d%d%d%d", &x1, &y1, &x2, &y2, &c);
		ar[x1][y1] += c;
		ar[x1][y2 + 1] -= c;
		ar[x2 + 1][y1] -= c;
		ar[x2 + 1][y2 + 1] += c;
	}
	for (int i = 1; i <= n; ++i)
	{
		for (int j = 1; j <= m; ++j)
		{
			ar[i][j] += ar[i - 1][j] + ar[i][j - 1] - ar[i - 1][j - 1];
			printf("%d ", ar[i][j] + origin[i][j]);
		}
		printf("\n");
	}
	return 0;
}
